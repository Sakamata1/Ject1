local function searchPartsForMainRoom()
    for _, part in ipairs(zones:GetChildren()) do
        if part:IsA("BasePart") then
            local isMainRoomValue = part:FindFirstChild("IsMainRoom")
            if isMainRoomValue and isMainRoomValue:IsA("BoolValue") and isMainRoomValue.Value == true then

            end

Tab:AddButton({
    Name = "Esp Ghost",
    Callback = function()
    -- Services
local RunService = game:GetService("RunService")
local PlayersService = game:GetService("Players")
local Workspace = game:GetService("Workspace")

-- Variables
local modelName = "Ghost" -- Name of the model to track
local BillboardGui
local Camera = game.Workspace.CurrentCamera
local ModelFound = false

-- Function to create BillboardGui on HumanoidRootPart of the model
local function createBillboardGui(model)
    BillboardGui = Instance.new("BillboardGui")
    BillboardGui.Size = UDim2.new(5, 0, 5, 0) -- Adjust size as needed
    BillboardGui.Adornee = model.HumanoidRootPart
    BillboardGui.StudsOffset = Vector3.new(0, 5, 0) -- Offset the BillboardGui slightly above the model
    BillboardGui.Parent = model.HumanoidRootPart

    local espTextLabel = Instance.new("TextLabel")
    espTextLabel.Size = UDim2.new(1, 0, 1, 0)
    espTextLabel.Text = modelName
    espTextLabel.TextColor3 = Color3.new(1, 0, 0)
    espTextLabel.TextScaled = true
    espTextLabel.Parent = BillboardGui
end

-- Function to update BillboardGui position and visibility
local function updateEsp(model)
    if model and model:IsA("Model") and model:FindFirstChild("HumanoidRootPart") then
        if not BillboardGui then
            createBillboardGui(model)
        else
            BillboardGui.Adornee = model.HumanoidRootPart
        end
        BillboardGui.Enabled = true
        ModelFound = true
    else
        ModelFound = false
        if BillboardGui then
            BillboardGui:Destroy()
            BillboardGui = nil
        end
    end
end

-- Main loop to continuously check for the model
local function mainLoop()
    while true do
        local model = Workspace:FindFirstChild(modelName)
        updateEsp(model)
        wait(1) -- Adjust the wait time as needed
    end
end

-- Connect the main loop to RunService
RunService.RenderStepped:Connect(function()
    if not ModelFound then
        mainLoop()
    end
end)
    end
})
